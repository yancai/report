docker 安装及配置

# 安装docker
sudo apt-get remove docker docker-engine docker.io

sudo apt-get update

sudo apt-get install apt-transport-https ca-certificates curl wget unzip zsh git software-properties-common

curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -

sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu xenial stable"

sudo apt-get install docker-ce=17.03.3~ce-0~ubuntu-xenial

# 安装好docker之后使用如下命令查看docker状态，状态为Active则说明docker服务正常
sudo systemctl status docker

# 配置 docker
sudo vim /etc/docker/daemon.json

添加如下内容

{
    "registry-mirrors": [
        "https://registry.docker-cn.com"
    ],
    "dns":[
        "114.114.114.114",
        "8.8.8.8",
        "8.8.4.4"
    ]
}

重启docker

sudo systemctl restart docker

sudo docker pull nginx

# 清理所有docker数据
sudo docker system prune -a

配置网络

sudo vim /etc/netplan/50-cloud-init.yaml

修改后的内容示例如下：

network:
    version: 2
    ethernets:
        ens32:
            # 注意修改ip
            addresses: [192.168.xxx.xxx/24]
            # 注意修改网关，可以使用 route -n 查询
            gateway4: 192.168.74.2
            nameservers:
                addresses: [127.0.0.53,114.114.114.114,8.8.8.8,8.8.4.4]

修改网络配置后使用如下命令应用配置：

sudo netplan apply

安装docker-compose

sudo curl -L https://github.com/docker/compose/releases/download/1.21.0/docker-compose-Linux-x86_64 -o /usr/local/bin/docker-compose

sudo chmod +x /usr/local/bin/docker-compose

安装rancher

mkdir rancher
vim docker-compose.yaml

rancher2.x的docker-compose.yaml内容如下：

version: '2.2'
services:
    rancher:
        image: rancher/rancher:v2.1.1
        container_name: rancher
        hostname: rancher
        environment:
            - TZ=Asia/Shanghai
        ports:
            - 80:80
            - 443:443
        restart: always

rancher1.x的docker-common.yaml内容如下：

version: '2.2'
services:
    rancher:
        image: rancher/server:v1.6.23
        container_name: rancher
        hostname: rancher
        environment:
            - TZ=Asia/Shanghai
        ports:
            - 8080:8080
        restart: always
